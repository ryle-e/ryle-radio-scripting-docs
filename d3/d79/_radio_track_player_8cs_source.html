151 /// <summary>
152 /// Updates the \ref sampleIncrement variable to match the current track
153 /// </summary>
154 public void UpdateSampleIncrement()
155 {
156 // we need to have a float variable for sampleIncrement as the sample rate of the track and the sample rate of the output may be different
157 // if they're different, it means that incrememnting Progress by 1 will make this track sound faster or slower depending on its SampleRate
158 // in order to counteract this, we set up a specific increment as the ratio between these two sample rates
159 //
160 // let's say that the Output's sampleRate is 44100, but the track's sample rate is 48000
161 // if we incremented Progress by 1 every sample, then, the track would sound slightly slower than usual as we're using the wrong SampleRate
162 // if we use this method though, we get an increment of ~0.92
163 // if we increment it by 0.92 every sample, then, the track sounds to be playing at the correct speed!
164 //
165 // this can get a bit confusing when we use Progress to get the sample INDEX, though, as we can't get an index with a float
166 // the solution here is just to convert it to an int, as seen in GetSample below
167 // this means that instead of skipping samples, it simply repeats the same one until the increment reaches the next sample
168 // the samples are happening so quickly that this repeat is completely unnoticeable- as far as i can tell, this is how all audio
169 // software and programs operate. as such, we use it here :)
170 //
171 // note: this explanation was mostly for my own future reference if i forget how this works lol
172 sampleIncrement = (decimal)TrackW.SampleRate / (decimal)baseSampleRate;
173
174 //if (TrackW.id == "music_old2")
175 // Debug.Log(TrackW.SampleCount);
176 }

178 /// <summary>
179 /// Gets the current sample from the track according to playback. <i>I would recommend reading the code comments for this method as they explain how the entire sample playback and evaluation process works</i>
180 /// </summary>
181 /// <param name="_tune">The tune value of the Output</param>
182 /// <param name="_receiverPosition">The position of the Output</param>
183 /// <param name="_otherVolume">The sum of the samples of previous tracks, according to the order in \ref RadioData.<br><b>See: </b>\ref RadioTrack.attenuation</param>
184 /// <param name="_outVolume">The volume of this sample to be added to `_otherVolume`</param>
185 /// <param name="_applyVolume">Whether or not Volume (`tune power * broadcast power * insulation`) should be applied</param>
186 /// <returns>The current sample</returns>
187 public float GetSample(float _tune, Vector3 _receiverPosition, float _otherVolume, out float _outVolume, bool _applyVolume = true)
188 {
189 // if this track is paused, return silence
190 if (isStopped || Paused)
191 {
192 _outVolume = 0f;
193 return 0;
194 }
195
196 // the output volume of this track right now
197 float volume = 0;
198 float gain = 100;
199
200 if (_applyVolume)
201 {
202 // long explanation of each method is in here so this can be kind of a core script for documentation
203
204 // get the gain of the track- this is a variable assigned in the inspector that serves as a basic increase to the
205 // loudness of the track without affecting attenuation or other audio values
206 // [[ this value is scaled down to ones in the property, so the below values refer to GainDisplay ]]
207 // if the gain is at 100, it will be at the default loudness
208 // if the gain is at 200, it will be double the loudness
209 // if the gain is at 50, it will be half the loudness
210 gain = TrackW.Gain;
211 OnGain(this, gain);
212
213 // get the tunePower of the track- this is a combination of the track's tuning power and attenuation
214 // the tuning power is defined by how closely the Output is tuned to this track. e.g, if this track's range is 100 - 300, and the
215 // Output's tune is 200, the track will likely be very loud- but if the tune is 120, it will be very quiet- the amount of loudness or
216 // quietness defined by the tune is named tune power here
217 //
218 // as well as the tuning power, this value is created with attenutation: how much quieter this track is when there are others playing
219 // the _otherVolume variable is the volume of other samples so far- if the attenuation of the track is high, then the tunePower will be
220 // lower depending on how high the other tracks' calculated tunePower is
221 //
222 // for more info, check inside the GetTunePower method
223 float tunePower = TrackW.GetTunePower(_tune, _otherVolume);
224 OnTunePower(this, tunePower);
225
226 // get the broadcast power of the track- this is dependent on where the Output is in relation to any RadioBroadcasters in the scene
227 // if the range of a Broadcaster is 100 units, and the output is 5 units away from it, it will likely hear the track loudly
228 // if the output is 95 units away, though, it will be heard quietly
229 // this works with many broadcasters and outputs- see RadioBroadcaster for more info
230 float broadcastPower = GetBroadcastPower(_receiverPosition);
231 OnBroadcastPower(this, broadcastPower);
232
233 // get the insulation of the track- this is effectively the inverse of broadcastPower, and is dependent on the position of the Output
234 // if the Insulator is a box 100 units wide, and the output is outside of it- the insulationMultiplier will be set to 1, as the track
235 // is not being insulated at all. if the output is inside the box, though, insulationMultiplier will be < 1 depending on the power in
236 // the RadioInsulator script.
237 // this was introduced to simulate "dead zones", or areas in which a broadcast can't be heard. if your output is inside an Insulator,
238 // the track will sound quieter- hence a lower insulationMultiplier the stronger the insulation
239 float insulationMultiplier = GetInsulation(_receiverPosition);
240 OnInsulation(this, insulationMultiplier);
241
242 // combine the broadcast power, tune power and insulation multiplier to the unified volume
243 volume = tunePower * broadcastPower * insulationMultiplier;
244 OnVolume(this, volume);
245 }
246
247 // get the sample at this moment from the track, and apply the volume and tunePower to it
248 float sample = TrackW.GetSample((int)Progress) * gain * volume;
249
250 // give back the volume to the Output
251 _outVolume = volume;
252
253 // and return the sample
254 return sample;
255 }

257 /// <summary>
258 /// Move this player to the next sample
259 /// </summary>
260 public void IncrementSample()
261 {
262 // if this track can't be played, don't increment its sample
263 if (isStopped || Paused)
264 return;
265
266 // the normal method if incrementing the sample if the track hasn't ended
267 void NormalIncrement()
268 {
269 OnSample.Invoke(this); // invoke the sample callback
270
271 // increment progress with the sampleIncrement, and clamp it between 0 and the maximum sample count
272 // see UpdateSampleIncrement() for more info on the sampleIncrement
273 Progress = Math.Clamp(Progress + sampleIncrement, 0, TrackW.SampleCount - 1);
274
275 UpdateSampleIncrement();
276 }
277
278 switch (PlayType)
279 {
280 // if this player loops when the track ends,
281 case PlayerType.Loop:
282 if (ProgressFraction >= 1) // if the track has ended,
283 {
284 OnEnd.Invoke(this); // invoke the end callback
285 Progress = 0; // reset the progress to 0, i.e loop the track
286 OnPlay.Invoke(this); // invoke the play callback as the track has been restarted
287 }
288 else // if the track is still partway playing,
289 {
290 NormalIncrement(); // increment normally
291 }
292
293 break;
294
295 // if this player only plays once,
296 case PlayerType.OneShot:
297 if (ProgressFraction >= 1) // if the track has ended,
298 {
299 OnEnd.Invoke(this); // invoke the end callback
300 Stop(); // stop and destroy this track
301 }
302 else
303 {
304 NormalIncrement(); // increment normally
305 }
306
307 break;
308 }
309 }

345 /// <summary>
346 /// Gets the broadcast power of this player using the position of the Output and any any \ref RadioBroadcaster
347 /// </summary>
348 /// <param name="_receiverPosition">The position of the Output</param>
349 /// <returns>The broadcast power- higher the closer the Output is to broadcasters</returns>
350 public float GetBroadcastPower(Vector3 _receiverPosition)
351 {
352 // if this track is forced to be global, it is forced to not use broadcasters- as such, its broadcast power is always 1
353 if (TrackW.forceGlobal)
354 return 1;
355
356 float outPower = 0;
357
358 // if this track has broadcasters,
359 if (TrackW.broadcasters.Count > 0)
360 {
361 foreach (RadioBroadcaster broadcaster in TrackW.broadcasters) // get the power from each broadcaster
362 outPower = Mathf.Max(outPower, broadcaster.GetPower(_receiverPosition)); // select the highest broadcast power and use that
363
364 // short discussion- this initially just added the broadcast power of each broadcaster together and used that- but after some thinking
365 // and research, i've changed it to use the maximum power overall to be closest to real-world function
366 // in real life, if we put too weak broadcasters next to each other with a receiver in the middle, the receiver doesn't get the combined
367 // signal of both broadcasters- it just gets two weak signals, from which is chooses the strongest. as such, we recreate this here-
368 // combining the power of the signals just wouldn't make sense from a practical standpoint
369 //
370 // i will note, however, that i was debating making this a modifiable option, like a BroadcasterCombineType package preference or
371 // something, but for the time being we're just using the one method here
372 }
373 // if this track has no broadcasters (and is therefore global),
374 else
375 outPower = 1; // the broadcast power is 1
376
377 // return the broadcast power, which should only be from 0 to 1
378 return Mathf.Clamp01(outPower);
379 }

381 /// <summary>
382 /// Gets the insulation multiplier of this player using the position of the output and the bounds of any \ref RadioInsulator
383 /// </summary>
384 /// <param name="_receiverPosition">The position of the Output</param>
385 /// <returns>The insulation multiplier- the more insulated the Output is, the lower the multiplier</returns>
386 public float GetInsulation(Vector3 _receiverPosition)
387 {
388 // if there are no insulators, the multiplier is 1
389 float outGain = 1;
390
391 foreach (RadioInsulator insulator in TrackW.insulators) // get the insulation power from each insulator
392 outGain -= insulator.GetPower(_receiverPosition); // subtract that power from the insulation multiplier
393
394 // shorter discussion- we use the opposite method to broadcast power here, that is combining all the insulation into one value- this makes
395 // more sense to me than just choosing the maximum- i imagine if we put two lead blocks in front of a receiever, the signal will be reduced
396 // by the combined insulation of both blocks, not just the biggest one
397 // (if this is incorrect, please tell me so i don't sound too stupid :(((( )
398
399 // return the insulation multiplier, which can only be from 0 to 1
400 return Mathf.Clamp01(outGain);
401 }
